# Plane-Alert candidate filters
#
# Format:
#   ICAO:<glob-pattern>:<owner fallback>
#   CALLSIGN:<glob-pattern>:<owner fallback>
#
# Matching logic is OR:
#   a record is considered if ICAO matches any ICAO pattern OR callsign matches any CALLSIGN pattern.
#
# Notes:
# - Matching is case-insensitive (internally uppercased)
# - Patterns use shell-style globs (* and ?)
#

# United States military (DoD; commonly used overall block)
ICAO:AE*:United States Military

# United States (non-civil US allocation; includes many US military/public aircraft—broad)
ICAO:AF*:United States Military

# United States military (DoD sub-block within AD* commonly observed)
ICAO:ADF[89A-F]*:United States Military

# United Kingdom military (commonly RAF/RN/RM; “43C–43F” widely associated with UK mil use)
ICAO:43[C-F]*:Royal Air Force

# Belgium military (commonly Belgian Air Component)
ICAO:44F*:Belgian Air Component

# Austria military (Austrian Armed Forces / air assets)
ICAO:44[4-7]*:Austrian Air Force
